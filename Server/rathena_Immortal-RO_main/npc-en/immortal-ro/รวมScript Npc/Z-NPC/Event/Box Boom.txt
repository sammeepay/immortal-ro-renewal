Gm:

exit:

OnInit:

end;

function	script	boxboom	{
mapannounce ""+$MapBB$+"","[Box BOom] : กล่อง มรณะ ได้ปรากฏขึ้นมาแล้ว !!! ",0,0x0DFF0D;

mapannounce ""+$MapBB$+"","[Box BOom] : โปรดคิดก่อนจะเปิดแต่ละกล่อง !!!",0,0x0DFF0D;

areamonster ""+$MapBB$+"",88,111,111,88,"B O X B O O M",1334,50,"BOXOPEN::OnOpen_Boom";
areamonster ""+$MapBB$+"",88,111,111,88,"B O X B O O M",1334,5,"BOXOPEN::OnOpen_C4";
areamonster ""+$MapBB$+"",88,111,111,88,"B O X B O O M",1334,10,"BOXOPEN::OnOpen_Ok";
return;
}

-	script	BOXOPEN	-1,{

OnOpen_Boom:
set .@Rate,rand(100);
if(.@Rate < 10) goto Chock;
mapannounce ""+$MapBB$+"","[Box BOom] : * "+strcharinfo(0)+" * เปิดเจอระเบิด ถูกส่งกลับบ้านเกิด !!!",0,0x0DFF0D;
specialeffect2 183;
sleep2 1000;
if(getmapusers(""+$MapBB$+"") <= 1) goto lastplayer;
warp "SavePoint",0,0;
end;

Chock:
mapannounce ""+$MapBB$+"","[Box BOom] : * "+strcharinfo(0)+" * เปิดเจอระเบิด เฮ้อ รอดไป ระเบิดด้าน !!!",0,0x0DFF0D;
specialeffect2 147;
warp "SavePoint",0,0;
end;

OnOpen_C4:
mapannounce ""+$MapBB$+"","[Box BOom] : * "+strcharinfo(0)+" * เปิดเจอระเบิด C4 แย่ละ !!!",0,0x0DFF0D;
donpcevent "LandEffect1::OnBoom";
donpcevent "LandEffect2::OnBoom";
donpcevent "LandEffect3::OnBoom";
donpcevent "LandEffect4::OnBoom";
donpcevent "LandEffect5::OnBoom";
killmonsterall ""+$MapBB$+"";
sleep2 2000;
mapannounce ""+$MapBB$+"","[Box BOom] : หว๊ออออ C4 ระเบิด ตาย ยกแมป !!!",0,0x0DFF0D;
killmonsterall ""+$MapBB$+"";
mapwarp ""+$MapBB$+"","morocc",0,0;
end;

OnOpen_Ok:
mapannounce ""+$MapBB$+"","[Box BOom] : * "+strcharinfo(0)+" * เป็นผู้โชคดีเปิดเจอ กล่อง สมบัติ !!!",0,0x0DFF0D;
getitem $ibb,$abb;
dispbottom "คุณได้รับ "+$ibb+" 300 Poring Coin.";
end;

lastplayer:
sleep2 1000;
announce " !!! ... กิจกรรม [Box BOom] จบลงแล้ว ... !!!",0,0x0DFF0D;
killmonsterall ""+$MapBB$+"";
warp "SavePoint",0,0;
mapwarp ""+$MapBB$+"","morocc",0,0;
end;
}
