-	script	PlayBoxTimer	-1,{
OnClock0002:
OnClock0102:
OnClock0202:
OnClock0302:
OnClock0402:
OnClock0502:
OnClock0602:
OnClock0702:
OnClock0802:
OnClock0902:
OnClock1002:
OnClock1102:
OnClock1202:
OnClock1302:
OnClock1402:
OnClock1502:
OnClock1602:
OnClock1702:
OnClock1802:
OnClock1902:
OnClock2002:
OnClock2102:

OnBox:
	getitem 29000,15;
	set .@rand, rand(10);
	setarray .@level[0],28000,28001,28002,28003,28004,28005,28006,28007;
	set .@runa, .@level[rand(getarraysize(.@level))];
	getitem .@runa,1; //>> จำนวนของ
	set $@PB_Count,$@PB_Count+1;
	announce "[ PlayBox ] : กล่องใบที่ [ "+$@PB_Count+"/100 ] ถูกพบโดยคุณ [ "+strcharinfo(0)+" ]",0;
end;
OnTimer60000:
if($@PB_Count==100){ stopnpctimer; set $@PB_Count,0; }
announce "[ PlayBox ] :  มีเวลาในการกำจัดอีก 4 นาที  ",0;
end;
OnTimer120000:
if($@PB_Count==100){ stopnpctimer; set $@PB_Count,0; }
announce "[ PlayBox ] :  มีเวลาในการกำจัดอีก 3 นาที  ",0;
end;
OnTimer180000:

end;
OnTimer240000:

end;
