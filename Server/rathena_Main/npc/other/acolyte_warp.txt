prontera,98,121,4	script	Acolyte#Prtclear	90,{
	mes "[Keiki]";
	mes "สวัสดีนักผจญภัย";
	mes "ฉันได้ศึกษาเวทมนตร์จากทั่วรูน - มิดการ์ทเพื่อยกระดับสิ่งที่ฉันเชื่อว่าเป็นหนึ่งในทักษะที่ยอดเยี่ยมที่สุดสำหรับคลาสฝึกหัด";
	next;

	if (is_function("VIP_iRO_Acolyte"))
		callfunc "VIP_iRO_Acolyte","[Keiki]","Juno";

	mes "[Keiki]";
	mes "ฉันเป็น Acolyte เพียงหนึ่งเดียวเท่านั้นที่ได้รับทักษะ Level 10 Warp Portal!";
	emotion ET_THINK, playerattached();
	next;
	mes "[Keiki]";
	mes "ถูกตัอง! และ...";
	mes "ฉันสัญญากับคุณว่าฉันจะไม่ลืมสถานที่ที่ฉันจำได้ไปแล้ว";
	mes "อยู่มาวันหนึ่งฉันจะเพิ่มทักษะของฉันในการวาร์ปไปที่ใดก็ได้ที่ฉันต้องการ ~";
	next;
	mes "[Keiki]";
	mes "ฉันยินดีที่จะแปรปรวนคุณไปยังสถานที่ต่างๆที่ฉันจำได้ด้วยค่าธรรมเนียมเล็กน้อย";
	mes "คุณต้องการใช้บริการนี้หรือไม่?";
	next;
	switch(select("ใช่:ไม่")) {
	case 1:
		mes "[Keiki]";
		mes "คุณอยากไปที่ไหน";
		mes "ฉันขอให้คุณโชคดีในการเดินทางของคุณ";
		next;

		setarray .@towns$[0], "Izlude", "Geffen", "Payon", "Morocc", "Alberta", "Al De Baran", "Comodo", "Umbala", "Juno";
		setarray .@cost[0],        600,     1200,    1200,     1200,      1800,          2200,     2200,     2200,   1800;

		set .@size, getarraysize(.@towns$);
		for(set .@i,0; .@i<.@size; set .@i,.@i+1)
			set .@menu$, .@menu$+.@towns$[.@i]+" -> "+.@cost[.@i]+"z:";
		set .@i, select(.@menu$+"ยกเลิก")-1;
		if (.@i == .@size)
			close;
		if (Zeny < .@cost[.@i]) {
			mes "[Keiki]";
			mes "ฉันขอโทษ แต่คุณไม่มี";
			mes "zeny เพียงพอสำหรับบริการเทเลพอร์ต";
			mes "ค่าเทเลพอร์ต";
			mes "ไป "+.@towns$[.@i]+" = "+.@cost[.@i]+" zeny.";
			close;
		}
		specialeffect2 EF_READYPORTAL;
		specialeffect2 EF_TELEPORTATION;
		specialeffect2 EF_PORTAL;
		next;
		set Zeny, Zeny - .@cost[.@i];
		switch(.@i) {
			case 0: if (checkre(0)) warp "izlude",128,98; else warp "izlude",91,105; break;
			case 1: warp "geffen",120,39; break;
			case 2: warp "payon",161,58; break;
			case 3: warp "morocc",156,46; break;
			case 4: warp "alberta",117,56; break;
			case 5: warp "aldebaran",168,112; break;
			case 6: warp "comodo",209,143; break;
			case 7: warp "umbala",100,154; break;
			case 8: warp "yuno",158,125; break;
		}
		close;
	case 2:
		close;
	}
}
