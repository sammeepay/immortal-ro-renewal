	mes "[Mage Guildsman]";
	mes "คุณต้องการทางออกหรือไม่อืมให้ 50 Zeny กับ Empty Test Tube 1 อัน";
	next;
	if (select("ตกลงตกลง:ไม่ลืมมันซะ") == 1) {
		mes "[Mage Guildsman]";
		if (Zeny < 50) {
			mes "เฮ้! เจ้ามีเงินไม่พอที่จะจ่าย 50 Zeny ของฉัน";
			close;
		}
		if (countitem(1092) == 0) {
			mes "เจ้าไม่สามารถพกสารละลายได้โดยไม่ต้องใช้ขวดเลยนำ Empty Test Tube มาให้ข้าด้วย";
			close;
		}
		delitem 1092,1; //Empty_Cylinder
		set Zeny, Zeny-50;
		getitem 1089,1; //Payon_Potion
	}
	close;
}
