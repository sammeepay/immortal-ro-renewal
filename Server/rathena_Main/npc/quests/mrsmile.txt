		mes "ด้วยกิจกรรมนี้ทุกคนจะมีความสุขและรอยยิ้ม";

		if (countitem(909) > 9 && countitem(914) > 9 && countitem(705) > 9) {

			mes "ขอแสดงความยินดีด้วย!";
