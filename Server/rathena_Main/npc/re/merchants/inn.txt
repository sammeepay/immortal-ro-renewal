// Brasilis
//============================================================
bra_in01,27,24,3	script	Hotel Keeper#bra1	478,{
	mes "[Hotel Keeper]";
	mes "ยินดีต้อนรับสู่โรงแรม Brasilis ที่สวยงาม";
	next;
	switch(select("เซฟ:นอนพัก -5000 zeny")) {
	case 1:
		mes "[Hotel Keeper]";
		mes "คุณต้องการบันทึกที่นี่ที่ Brasilis Hotel หรือไม่";
		next;
		switch(select("ไม่ขอบใจนะ:แน่นอน")) {
		case 1:
			mes "[Hotel Keeper]";
			mes "โอเคสนุกไปกับการพัก";
			close;
		case 2:
			mes "[Hotel Keeper]";
			mes "การตอบกลับของคุณได้รับการบันทึกที่นี่ที่โรงแรมฉันหวังว่าคุณจะสนุกกับการพักที่นี่ใน Brasilis";
			savepoint "bra_in01",144,69,1,1;
			close;
		}
	case 2:
		if (Zeny > 4999) {
			mes "[Hotel Keeper]";
			mes "ฉันจะแสดงให้คุณเห็นห้องที่ดี";
			close2;
			set Zeny, Zeny-5000;
			percentheal 100,100;
			warp "bra_in01",144,69;
			end;
		}
		else {
			mes "[Hotel Keeper]";
			mes "ฉันขอโทษ แต่ค่าบริการ 5,000 zeny ต่อคืน";
			close;
		}
	}
}
