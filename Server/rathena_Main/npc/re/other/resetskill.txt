	mes "[Hypnotist]";
	mes "สวัสดีนักผจญภัย";
	mes "ฉันเป็นสมาชิกของ Hypnotist";
	mes "Academy ส่งมาที่ Izlude";
	mes "เพื่อให้บริการ Skill Reset";
	mes "สำหรับตัวละคร First Class ในราคาที่คุ้มค่า: ฟรี!";
	next;
	mes "[Hypnotist]";
	mes "ถึงแม้ว่าฉันจะเสนอให้ไม่ จำกัด ";
	mes "รีเซ็ต stat / skill สำหรับตอนนี้ฉันมี";
	mes "สองเงื่อนไขที่ต้องเป็น";
	mes "เติมเต็มก่อนอื่นคุณต้องเป็น";
	mes "ต่ำกว่า ^FF0000Base Level 50^000000.";
	next;
	mes "[Hypnotist]";
	mes "อย่างที่สองคุณต้องเป็น";
	mes "^FF0000Swordman, Acolyte, Mage,";
	mes "Thief, Archer, Merchant,";
	mes "Taekwon Boy, หรือ Taekwon Girl,";
	mes "Gunslinger และ Ninja^000000";
	mes "ตัวละครอาชีพที่ต้องมีคุณสมบัติ";
	mes "คุณมีคำถามหรือไม่?";
	next;
	switch(select("^FF0000รีเซ็ต Stat/Skill?:รีเซ็ต stats:รีเซ็ต skills^000000:ยกเลิก")) {
	case 1:
		mes "[Hypnotist]";
		mes "รีเซ็ต Stat / Skill อนุญาตให้การผจญภัยแจกจ่ายคะแนนทักษะของตนอีกครั้งหากพวกเขาไม่พอใจกับทักษะปัจจุบัน";
		next;
		mes "[Hypnotist]";
		mes "ก่อนที่จะดำเนินการรีเซ็ตค่าสถานะ / ทักษะคุณต้องลดน้ำหนักของสิ่งของที่บรรทุกทั้งหมดของตัวละครของคุณให้เหลือ 0";
		mes "คุณสามารถใส่ไอเทมพิเศษใน Kafra Storage ได้";
		next;
		mes "[Hypnotist]";
		mes "อ่า, ถอนการติดตั้ง ^FF0000รถเข็น^000000 ของคุณก่อน";
		mes "ถ้าคุณมีอุปกรณ์";
		mes "มิฉะนั้นการสะกดจิตจะไม่ได้ผลหรือจะย้อนกลับ... ";
		close;
	case 2:
		set .@str$,"Stat";
	case 3:
		if (.@str$ == "")
			set .@str$,"Skill";
		mes "[Hypnotist]";
		mes "คุณแน่ใจหรือว่าต้องการดำเนินการกับบริการ ^FF0000รีเซ็ต " + .@str$ + "^000000";
		next;
		if(select("Yes:Cancel") == 1)
			break;
	case 4:
		mes "[Hypnotist]";
		mes "ขอบคุณและขอให้โชคดีกับการผจญภัยของคุณ";
		mes "โปรดเดินทางโดยสวัสดิภาพ ~";
		close;
	}
	if ((Class >= Job_Swordman && Class <= Job_Thief) || Class == Job_Taekwon || Class == Job_Gunslinger || Class == Job_Ninja) {
		if (checkcart()) {
			mes "[Hypnotist]";
			mes "อ้อ! กรุณาถอดรถเข็นของคุณออกก่อนดำเนินการต่อด้วยบริการรีเซ็ต " + .@str$ + ".";
			mes "ขอบคุณที่ร่วมมือ ~";
			close;
		}
		if (Weight != 0) {
			mes "[Hypnotist]";
			mes "ถ้าคุณมาที่นี่เพื่อบริการรีเซ็ต " + .@str$ + " ของฉัน โปรดจำไว้ว่าคุณไม่สามารถรีเซ็ต " + .@str$ + " ของคุณได้จนกว่า ^FF0000น้ำหนักในกระเป๋าของคุณลดเหลือ 0^000000";
			next;
			mes "[Hypnotist]";
			mes "ทำไมคุณไม่วางของลงใน Kafra Storage ซะก่อนล่ะวิธีนี้คุณจะสามารถเก็บสินค้าทั้งหมดของคุณได้อย่างปลอดภัย";
			close;
		}
		if (BaseLevel >= 50 && (Class == Job_Taekwon || Class == Job_Gunslinger || Class == Job_Ninja)) {
			mes "[Hypnotist]";
			mes "ฉันขอโทษ แต่ตัวละครที่มีเลเวลพื้นฐานสูงกว่า 50 จะไม่มีสิทธิ์ได้รับบริการรีเซ็ต" +.@str$ + " ที่ฉันมีให้";
			close;
		}
		mes "[Hypnotist]";
		mes "ขอบคุณที่ใช้บริการ "+.@str$+" Redistribution ของฉัน";
		mes "โอ้ขอให้โชคดีกับการเดินทางของคุณนักผจญภัย";
		if (.@str$ == "Stat")
			resetstatus;
		else
			resetskill;
		close;
	} else {
		mes "[Hypnotist]";
		mes "I'm sorry, but your";
		mes "Job Class doesn't qualify";
		mes "for the "+.@str$+" Reset service";
		mes "that I provide. I can only";
		mes "offer "+.@str$+" Resets to the";
		mes "ฉันขอโทษ แต่ Job Class ของคุณไม่มีคุณสมบัติสำหรับบริการรีเซ็ต  "+.@str$+" ที่ฉันมีให้ฉันสามารถเสนอรีเซ็ต "+.@str$+" ให้กับอาชีพต่อไปนี้เท่านั้น... ";
		next;
		mes "[Hypnotist]";
		mes "^FF0000Swordman, Acolyte,";
		mes "Mage, Thief, Archer,";
		mes "Merchant, Taekwon";
		mes "Boy, Taekwon Girl,";
		mes "Gunslinder และ Ninja^000000.";
		close;
	}
}
