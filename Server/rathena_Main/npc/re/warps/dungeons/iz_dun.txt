	mes "[Gatekeeper]";
	mes "หยุด!";
	mes "อะไรทำให้คุณมาที่นี่นักผจญภัย?";
	mes "เราค้นพบชั้นใหม่ของเมืองใต้น้ำนี้";
	next;
	mes "[Gatekeeper]";
	mes "คุณคิดว่าคุณจะท้าทายหรือไม่";
	mes "สัตว์ประหลาดที่นี่อยู่ระหว่างเลเวล 110 และ 124";
	mes "และเจ้าต้องระวัง Kraken!";
	next;
	switch(select("ปล่อย Kraken!: ฉันจะอยู่ที่นี่")) {
	case 1:
		mes "[Gatekeeper]";
		mes "จงระวังและโชคดี";
		mes "ถ้าคุณเห็น Kraken ต้องแน่ใจว่าคุณมีเพื่อนช่วยหรือทำในสิ่งที่เราทำและวิ่ง";
		close2;

		end;
	case 2:
		mes "[Gatekeeper]";
		mes "เป็นการเตรียมพร้อมที่ดีที่สุดก่อนที่จะสำรวจพื้นที่ใหม่";
		mes "กลับมาเมื่อเจ้าพร้อมที่จะไป";
		close;
	}
}
